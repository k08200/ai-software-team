1. GET THE APPLICATION URL

{{- if .Values.ingress.enabled }}
   https://{{ .Values.ingress.hostname }}/

   (DNS must point {{ .Values.ingress.hostname }} to your ingress-controller's external IP)

   Get the ingress-controller IP:
     kubectl get svc -n ingress-nginx ingress-nginx-controller \
       -o jsonpath='{.status.loadBalancer.ingress[0].ip}'

{{- else if eq .Values.service.type "NodePort" }}
   export NODE_PORT=$(kubectl get svc {{ include "ai-software-team.fullname" . }} \
     -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
   export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
   echo "http://$NODE_IP:$NODE_PORT"

{{- else }}
   # Port-forward for local access:
   kubectl port-forward svc/{{ include "ai-software-team.fullname" . }} 3001:{{ .Values.service.port }} \
     -n {{ .Release.Namespace }}

   # Then open:  http://localhost:3001

{{- end }}

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2. SET YOUR ANTHROPIC API KEY (if not set at install time)

   kubectl create secret generic {{ include "ai-software-team.fullname" . }}-secrets \
     --from-literal=anthropic-api-key="sk-ant-YOUR_KEY_HERE" \
     --from-literal=jwt-secret="$(openssl rand -hex 32)" \
     --from-literal=stripe-secret-key="" \
     --from-literal=database-url="postgresql://user:pass@host:5432/db" \
     -n {{ .Release.Namespace }} \
     --dry-run=client -o yaml | kubectl apply -f -

   Then restart the deployment to pick up the new secret:
     kubectl rollout restart deployment/{{ include "ai-software-team.fullname" . }} \
       -n {{ .Release.Namespace }}

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3. RUN YOUR FIRST PIPELINE

   # Check that the service is healthy:
   curl https://{{ .Values.ingress.hostname | default "localhost:3001" }}/health

   # Kick off a pipeline run (SSE stream):
   curl -N -X POST https://{{ .Values.ingress.hostname | default "localhost:3001" }}/api/pipeline/run \
     -H "Content-Type: application/json" \
     -d '{"idea": "A REST API for a simple todo list with PostgreSQL"}' \
     --no-buffer

   # List recent sessions:
   curl https://{{ .Values.ingress.hostname | default "localhost:3001" }}/api/pipeline/sessions

   # Download generated ZIP (replace SESSION_ID):
   curl -OJ https://{{ .Values.ingress.hostname | default "localhost:3001" }}/api/pipeline/download/SESSION_ID
